124 """
125 rem_vertex!(g, v)
126
127 Remove the vertex `v` from graph `g`. Return `false` if removal fails
128 (e.g., if vertex is not in the graph); `true` otherwise.
129
130 ### Performance
131 Time complexity is ``\\mathcal{O}(k^2)``, where ``k`` is the max of the degrees
132 of vertex ``v`` and vertex ``|V|``.
133
134 ### Implementation Notes
135 This operation has to be performed carefully if one keeps external
136 data structures indexed by edges or vertices in the graph, since
137 internally the removal is performed swapping the vertices `v` and ``|V|``,
138 and removing the last vertex ``|V|`` from the graph. After removal the
139 vertices in `g` will be indexed by ``1:|V|-1``.
140
141 # Examples
142 ```jldoctest
143 julia> using LightGraphs
144
145 julia> g = SimpleGraph(2);
146
147 julia> rem_vertex!(g, 2)
148 true
149
150 julia> rem_vertex!(g, 2)
151 false
152 ```
153 """
154 function rem_vertex!(g::AbstractSimpleGraph, v::Integer)
155 v in vertices(g) || return false
156 n = nv(g)
157 self_loop_n = false # true if n is self-looped (see #820)
158
159 # remove the in_edges from v
160 srcs = copy(inneighbors(g, v))
161 @inbounds for s in srcs
162 rem_edge!(g, edgetype(g)(s, v))
163 end
164 # remove the in_edges from the last vertex
165 neigs = copy(inneighbors(g, n))
166 @inbounds for s in neigs
167 rem_edge!(g, edgetype(g)(s, n))
168 end
169 if v != n
170 # add the edges from n back to v
171 @inbounds for s in neigs
172 if s != n # don't add an edge to the last vertex - see #820.
173 add_edge!(g, edgetype(g)(s, v))
174 else
175 self_loop_n = true
176 end
177 end
178 end
179
180 if is_directed(g)
181 # remove the out_edges from v
182 dsts = copy(outneighbors(g, v))
183 @inbounds for d in dsts
184 rem_edge!(g, edgetype(g)(v, d))
185 end
186 # remove the out_edges from the last vertex
187 neigs = copy(outneighbors(g, n))
188 @inbounds for d in neigs
189 rem_edge!(g, edgetype(g)(n, d))
190 end
191 if v != n
192 # add the out_edges back to v
193 @inbounds for d in neigs
194 if d != n
195 add_edge!(g, edgetype(g)(v, d))
196 end
197 end
198 end
199 end
200 if self_loop_n
201 add_edge!(g, edgetype(g)(v, v))
202 end
203 pop!(g.fadjlist)
204 if is_directed(g)
205 pop!(g.badjlist)
206 end
207 return true
208 end
