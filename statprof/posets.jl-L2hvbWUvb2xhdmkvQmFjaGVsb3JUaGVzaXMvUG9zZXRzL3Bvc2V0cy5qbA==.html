102 function isIso(g::SimpleDiGraph, s::SimpleDiGraph, a::Array{Int})
103 #Easy stuff first
104 if nv(g) != nv(s) || length(a) != nv(g)
105 return false
106 end
107 for i in 1:length(a) #Confirm that "a" is an isomorphism
108 if !(i in a)
109 return false
110 end
111 end
112 for vertex in 1:length(a)
113 neighbours_in_g = inneighbors(g, vertex)
114 neighbours_in_s = inneighbors(s, a[vertex])
115 for neighbour in neighbours_in_g
116 if !(a[neighbour] in neighbours_in_s)
117 return false
118 end
119 end
120 neighbours_out_g = outneighbors(g, vertex)
121 neighbours_out_s = outneighbors(s, a[vertex])
122 for neighbour in neighbours_out_g
123 if !(a[neighbour] in neighbours_out_s)
124 return false
125 end
126 end
127 end

131 """Check if the graphs g and s are isomorphic to each other"""
132 function isIso(g::SimpleDiGraph, s::SimpleDiGraph)
133 #Start with the easy stuff
134 n = nv(g)
135 if n != nv(s) || ne(g) != ne(s)
136 return false
137 end
138 #Make arrays of "graph invariants" e.g integer tuples that describe the nodes
139 g_v_profiles = Array{Tuple{Int, Int}}(undef, n)
140 s_v_profiles = Array{Tuple{Int, Int}}(undef, n)
141 for v in 1:n
142 g_v_profiles[v] = (inHash(g, v), outHash(g, v))
143 s_v_profiles[v] = (inHash(s, v), outHash(s, v))
144 end
145 # Check that the vertex profiles match in both graphs
146 used = zeros(Bool, n)
147 for i in 1:n
148 found = false
149 for j in 1:n
150 if g_v_profiles[i] == s_v_profiles[j] && !used[j]
151 found = true
152 break
153 end
154 end
155 if !found
156 return false
157 end
158 end
159 # Construct classes of vertices that can be isomorphic
160 v_classes = Array{Tuple{Array{Int}, Array{Int}}}(undef, 0)
161 indexed_g = zeros(Bool, n)
162 indexed_s = zeros(Bool, n)
163 for i in 1:n
164 if indexed_g[i]
165 continue
166 end
167 class = (Array{Int}(undef, 1), Array{Int}(undef, 0))
168 class[1][1] = i
169 for v in 1:n
170 if !indexed_s[v] && s_v_profiles[v] == g_v_profiles[i]
171 push!(class[2], v)
172 indexed_s[v] = true
173 end
174 if !indexed_g[v] && g_v_profiles[v] == g_v_profiles[i] && v != i
175 push!(class[1], v)
176 indexed_g[v] = true
177 end
178 end
179 push!(v_classes, class)
180 indexed_g[i] = true
181 end
182 isom_parts = Array{Array{Array{Tuple{Int, Int}}}}(undef, length(v_classes))
183 for i in 1:length(v_classes)
184 isom_parts[i] = []
185 lh = v_classes[i][1] # Nodes in g
186 rh = v_classes[i][2] # Nodes in s
187 for perm in permutations(rh)
188 push!(isom_parts[i], collect(zip(lh, perm)))
189 end
190 end
191 for perm in Iterators.product(isom_parts...)
192 pos_isom = Array{Int}(undef, n)
193 mapped = zeros(Bool, n)
194 # Now we use the vertex profiles to see if we can fill in pos_isom
195 #println(perm)
196 for info_array in perm
197 if !(false in mapped)
198 break
199 end
200 for mapping in info_array
201 pos_isom[mapping[1]] = mapping[2]
202 end
203 end
204 #println(perm)
205 #println(pos_isom)
206 if isIso(g, s, pos_isom)
207 return true
208 end
209 end
210 return false

213 """Generate all of the posets of size n, return them as a set"""
214 function genPosets(n::Int)
215 #Quickly handle special cases 0 and 1 because they're easy
216 if n == 0
217 return SimpleDiGraph(0)
218 elseif n == 1
219 return SimpleDiGraph(1)
220 end
221 #Movin on...
222 graph_counter = 0
223 max_edges = (n*(n-1))÷2 #÷ is the integer division character
224 representatives = Array{SimpleDiGraph}(undef, 0)
225 all_edges = Array{Tuple{Int, Int}}(undef, max_edges)
226 en::UInt = 1
227 for i in 1:(n-1)
228 for j in (i+1):n
229 all_edges[en] = (i, j)
230 en += 1
231 end
232 end
233 println(all_edges)
234 for numedges in 0:max_edges
235 #@printf("Considering %d edges\n", numedges)
236 #Get all the ways to distribute the edges
237 distributions = collect(combinations(all_edges, numedges))
238 ld = length(distributions)
239 for i in 1:ld
240 #@printf("Checking distribution %d/%d\n", i, ld)
241 g = SimpleDiGraph(n) #New graph with n nodes
242 #Now we put the edges in
243 for e in distributions[i]
244 add_edge!(g, e[1], e[2])
245 end
246 transitiveclosure!(g, true) #Take the transitive closure
247 if isAntiSymmetric(g)
248 is_iso = false
249 for s in representatives
250 if isIso(g, s)
251 is_iso = true
252 break
253 end
254 end
255 if is_iso
256 continue
257 end
258 graph_counter += 1
259 println("Found a representative, " * string(graph_counter) * " so far")
260 push!(representatives, g)
261 end
262 end
263 end
264 return representatives
265 end
